## Scorched
diplomacy_actor_is_not_scorched = {
	custom_tooltip = {
		fail_text = "requires_actor_not_scorched_earth"
		OR = {
			is_genocidal_infernal = no
			from = { is_infernal_empire = yes }
		}
	}
}
diplomacy_recipient_is_not_scorched = {
	if = {
		limit = { from = { is_genocidal_infernal = yes } }
		if = {
			limit = {
				has_intel = {
					who = from
					intel = civics
				}
			}
			custom_tooltip = {
				fail_text = "requires_recipient_not_scorched_earth"
				OR = {
					is_infernal_empire = yes
					from = { is_genocidal_infernal = no }
				}
			}
		}
		else = {
			custom_tooltip = {
				fail_text = diplo_action_no_low_intel
				OR = {
					is_genocidal_infernal = no
					from = { is_infernal_empire = yes }
				}
			}
		}
	}
}

## Pompous
# this = country, actor
# from = country, recipient
diplomacy_recipient_is_not_pompous = {
	if = {
		limit = {
			from = { has_valid_civic = civic_pompous_purists }
			NOT = { root = { is_overlord_to = from } }
		}
		if = {
			limit = {
				has_intel = {
					who = from
					intel = civics
				}
			}
			custom_tooltip = {
				fail_text = requires_recipient_not_pompous
				always = no
			}
		}
		else = {
			custom_tooltip = {
				fail_text = diplo_action_no_low_intel
				always = no
			}
		}
	}
}
diplomacy_actor_is_not_pompous = {
	custom_tooltip = {
		fail_text = "requires_actor_not_pompous"
		NOT = { has_valid_civic = civic_pompous_purists	}
	}
}
diplomacy_recipient_is_not_pompous_bypass = {
	if = {
		limit = {
			from = { has_valid_civic = civic_pompous_purists }
			NOT = { root = { is_overlord_to = from } }
			root = { is_same_value = prevprev } # Bypass check if we are proposing on behalf of recipient
		}
		if = {
			limit = {
				has_intel = {
					who = from
					intel = civics
				}
			}
			custom_tooltip = {
				fail_text = requires_recipient_not_pompous
				always = no
			}
		}
		else = {
			custom_tooltip = {
				fail_text = diplo_action_no_low_intel
				always = no
			}
		}
	}
}

diplomacy_treaties = {
	custom_tooltip = {
		fail_text = "federation_actor_no_separate_treaties"
		NOT = {
			AND = {
				has_federation = yes
				federation = { has_federation_law = treaties_separate_no }
				NOT = { federation = { any_member = { is_same_value = root.from } } }
			}
		}
	}
	custom_tooltip = {
		fail_text = "federation_recipient_no_separate_treaties"
		from = {
			NOT = {
				AND = {
					has_federation = yes

						has_federation_law = treaties_separate_no
					}
					NOT = { federation = { any_member = { is_same_value = root } } }
				}
			}
		}
	}

is_diplo_robot_empire = {
	OR = {
		owner_species = { has_trait = trait_mechanical }
		owner_species = { has_trait = trait_machine_unit }
		has_country_flag = synthetic_empire
		is_machine_empire = yes
		is_individual_machine = yes
	}
